   20 ------------------------------------------------------------------------------
   21 -- | Given a 'Response', return its body as a 'ByteString'.
   22 --
   23 -- Example:
   24 --
   25 -- @
   26 -- ghci> 'getResponseBody' 'emptyResponse'
   27 -- \"\"
   28 -- @
   29 --
   30 getResponseBody :: Response -> IO ByteString
   31 getResponseBody rsp = do
   32     (os, grab) <- Streams.listOutputStream
   33     enum os
   34     liftM toBS grab
   35 
   36   where
   37     enum os = do
   38         os' <- rspBodyToEnum (rspBody rsp) os
   39         Streams.write Nothing os'
   40 
   41     toBS = S.concat . L.toChunks . toLazyByteString . mconcat
   42

   45 -- | Given a 'Response', assert that its HTTP status code is 200 (success).
   46 --
   47 -- Example:
   48 --
   49 -- @
   50 -- ghci> :set -XOverloadedStrings
   51 -- ghci> import qualified "Test.HUnit" as T
   52 -- ghci> let test = T.runTestTT . T.TestCase
   53 -- ghci> test $ 'assertSuccess' 'Snap.Core.emptyResponse'
   54 -- Cases: 1  Tried: 1  Errors: 0  Failures: 0
   55 -- Counts {cases = 1, tried = 1, errors = 0, failures = 0}
   56 -- ghci> test $ 'assertSuccess' ('Snap.Core.setResponseStatus' 500 \"Internal Server Error\" 'Snap.Core.emptyResponse')
   57 -- ### Failure:
   58 -- Expected success (200) but got (500)
   59 -- expected: 200
   60 --  but got: 500
   61 -- Cases: 1  Tried: 1  Errors: 0  Failures: 1
   62 -- Counts {cases = 1, tried = 1, errors = 0, failures = 1}
   63 -- @
   64 assertSuccess :: Response -> Assertion
   65 assertSuccess rsp = assertEqual message 200 status
   66   where
   67     message = "Expected success (200) but got (" ++ (show status) ++ ")"
   68     status  = rspStatus rsp

   89 ------------------------------------------------------------------------------
   90 -- | Given a 'Response', assert that its HTTP status code is between 300 and 399
   91 -- (a redirect), and that the Location header of the 'Response' points to the
   92 -- specified URI.
   93 --
   94 -- Example:
   95 --
   96 -- @
   97 -- ghci> :set -XOverloadedStrings
   98 -- ghci> let r' = 'Snap.Core.setResponseStatus' 301 \"Moved Permanently\" 'Snap.Core.emptyResponse'
   99 -- ghci> let r  = 'Snap.Core.setHeader' \"Location\" \"www.example.com\" r'
  100 -- ghci> 'assertRedirectTo' \"www.example.com\" r
  101 -- ghci> 'assertRedirectTo' \"www.example.com\" 'Snap.Core.emptyResponse'
  102 -- *** Exception: HUnitFailure \"Expected redirect but got status code (200)\"
  103 -- @
  104 assertRedirectTo :: ByteString     -- ^ The Response should redirect to this
  105                                    -- URI
  106                  -> Response
  107                  -> Assertion
  108 assertRedirectTo uri rsp = do
  109     assertRedirect rsp
  110     assertEqual message uri rspUri
  111 
  112   where
  113     rspUri = fromJust $ getHeader "Location" rsp
  114     message = "Expected redirect to " ++ show uri
  115               ++ " but got redirected to "
  116               ++ show rspUri ++ " instead"
  117 
  118 
  119 ------------------------------------------------------------------------------
  120 -- | Given a 'Response', assert that its HTTP status code is between 300 and 399
  121 -- (a redirect).
  122 --
  123 -- Example:
  124 --
  125 -- @
  126 -- ghci> :set -XOverloadedStrings
  127 -- ghci> 'assertRedirect' $ 'Snap.Core.setResponseStatus' 301 \"Moved Permanently\" 'Snap.Core.emptyResponse'
  128 -- ghci> 'assertRedirect' 'Snap.Core.emptyResponse'
  129 -- *** Exception: HUnitFailure \"Expected redirect but got status code (200)\"
  130 -- @
  131 assertRedirect :: Response -> Assertion
  132 assertRedirect rsp = assertBool message (300 <= status && status <= 399)
  133   where
  134     message = "Expected redirect but got status code ("
  135               ++ show status ++ ")"
  136     status  = rspStatus rsp

  139 ------------------------------------------------------------------------------
  140 -- | Given a 'Response', assert that its body matches the given regular
  141 -- expression.
  142 --
  143 -- Example:
  144 --
  145 -- @
  146 -- ghci> :set -XOverloadedStrings
  147 -- ghci> import qualified "System.IO.Streams" as Streams
  148 -- ghci> import qualified "Data.ByteString.Builder" as Builder
  149 -- ghci> :{
  150 -- ghci| let r = 'Snap.Core.setResponseBody'
  151 -- ghci|         (\out -> do
  152 -- ghci|             Streams.write (Just $ Builder.byteString \"Hello, world!\") out
  153 -- ghci|             return out)
  154 -- ghci|         'Snap.Core.emptyResponse'
  155 -- ghci| :}
  156 -- ghci> 'assertBodyContains' \"^Hello\" r
  157 -- ghci> 'assertBodyContains' \"Bye\" r
  158 -- *** Exception: HUnitFailure "Expected body to match regexp \\\"\\\"Bye\\\"\\\", but didn\'t"
  159 -- @
  160 assertBodyContains :: ByteString  -- ^ Regexp that will match the body content
  161                    -> Response
  162                    -> Assertion
  163 assertBodyContains match rsp = do
  164     body <- getResponseBody rsp
  165     assertBool message (body =~ match)
  166   where
  167     message = "Expected body to match regexp \"" ++ show match
  168               ++ "\", but didn't"
